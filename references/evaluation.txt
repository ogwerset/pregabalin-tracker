    df[col] = pd.to_numeric(df[col], errors='coerce')

# Funkcja do ekstrakcji dawki z tekstu
def extract_dose(text):
    if pd.isna(text) or text in ['-', 'N/A', 'NIE', 'nie', 'BRAK']:
        return np.nan
    import re
    match = re.search(r'(\d+)MG', str(text).upper())
    if match:
        return int(match.group(1))
    return np.nan

# Ekstrakcja dawek

    if pd.isna(note) or note in ['-', 'BRAK']:
        return False
    return any(kw in str(note).lower() for kw in substance_keywords)

df['Uzywki_Today'] = df['Notatki'].apply(check_substances)

# Tworzenie flagi przesuniÄ™tej o 1 dzieÅ„ (wpÅ‚yw kaca/zjazdu)
